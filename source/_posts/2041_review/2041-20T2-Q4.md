截取后对着学号计数就可以了。

```python
#!/usr/bin/env python3

import sys
from collections import defaultdict

dic = defaultdict(int)

for line in sys.stdin:
    line = line.strip()
    if not line:
        continue
    field = line.split("|")
    id_and_name = field[1:3]

    dic[field[1]] += 1

ans = []
for key in dic:
    if dic[key] == 2:
        ans.append(key)

ans.sort()
for id in ans:
    print(id)
```
